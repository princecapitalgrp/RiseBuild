New Types

OperatingState — produced by InterpretationService
type OperatingMode =
  | 'RECOVERY'               // low energy, needs containment
  | 'FOCUS_REBUILD'          // moderate energy, clarity first
  | 'STEADY_EXECUTION'       // normal capacity, standard OS
  | 'HIGH_STAKES_DAY'        // important commitment ahead
  | 'OVERWHELM_CONTAINMENT'; // triage mode

interface OperatingState {
  schemaVersion: string;
  sessionType: SessionType;   // 'morning' | 'midday' | 'evening' | ...
  mode: OperatingMode;
  urgency: 'LOW' | 'MODERATE' | 'HIGH' | 'CRITICAL';
  capacity: number;           // 0-1, from energy + stress + sleep
  confidence: number;         // 0-1, from memory accuracy history
  producedAt: string;
  modelVersion: string;
}
PlanRationale — embedded in OperatingPlan
interface PlanRationale {
  noticed: string[];        // what the system observed (no user text)
  planLogic: string;        // why this plan shape was chosen
  protecting?: string;      // what this plan preserves
  creating?: string;        // what this plan builds toward
  operatingMode: OperatingMode;
}

// Example:
{
  noticed: [
    "Low capacity (0.3) — high stress, low energy",
    "High-stakes commitment in 90 minutes",
    "Self-doubt pattern active"
  ],
  planLogic: "Containment before optimization. Stability before performance.",
  protecting: "Cognitive reserve and nervous system stability",
  creating: "One clear first sentence as minimum viable anchor",
  operatingMode: "HIGH_STAKES_DAY"
}
TrustSignal — produced by TrustEvaluator, lives in MemorySummary
interface TrustSignal {
  schemaVersion: string;
  computedAt: string;
  sampleSize: number;
  // Components — all derived from execution + reflection data
  accuracyTrend: number;            // feltAccurate rolling avg, normalized
  followThroughConsistency: number; // % sessions: yes or partial
  abandonmentRate: number;          // lower = more trust (inverted)
  correctionFrequency: number;      // step skips/overrides (lower = trust)
  beginWillingness: number;         // session start vs wake time proximity
  // Composite
  composite: number;                // 0-1
  trend: 'BUILDING' | 'STABLE' | 'ERODING';
}
MemorySummary — upgraded behavioral model
interface MemorySummary {
  schemaVersion: string;
  engineVersion: string;
  updatedAt: string;
  updateCount: number;
  // How this person functions — behavioral buckets
  stabilizers: string[];         // conditions that reliably improve state
  derailers: string[];           // patterns that pull them off track
  tractionBuilders: string[];    // first moves that create momentum
  selfTrustThreats: string[];    // patterns that erode self-trust
  reliableFirstMoves: string[];  // moves that consistently start good sessions
  // Accuracy patterns (legacy, kept)
  lowAccuracyPatterns: string[];
  highAccuracyPatterns: string[];
  abandonmentPatterns: string[];
  // Quantitative
  averageFollowThrough: number;
  averageExecutionDepth: number;
  recentSessionCount: number;
  // Trust
  trustSignal: TrustSignal;
}
